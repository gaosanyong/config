Run below commands to link bash setup 

```
cd 
ln -s ~/.config/bash/bash_alias.sh .bash_alias
ln -s ~/.config/bash/bash_function.sh .bash_function
ln -s ~/.config/bash/bash_profile.sh .bash_profile
ln -s ~/.config/bash/bashrc.sh .bashrc
ln -s ~/.config/bash/inputrc.sh .inputrc
ln -s ~/.config/vim/vimrc .vimrc
ln -s ~/.config/tmux/tmux.conf .tmux.conf
```
